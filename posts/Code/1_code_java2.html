package ou2;
public class TriangleAndItsCircles{

  //basic computations
  public static double area (double A, double H) {
    return (A * H / 2);
  }

  public static double perimeter (double A, double B, double C) {
    return A + B + C;
  }


  //circles
  public static double rcircumcircle (double A, double B, double C) {
    double radius = (Math.sqrt(A * A * B * B * C *C /((A+B+C)*(B+C-A)*(C+A-B)*(A+B-C))));
    return radius;
  }

  public static double circumcircle (double A, double B, double C) {
    double radius = (Math.sqrt(A * A * B * B * C *C /((A+B+C)*(B+C-A)*(C+A-B)*(A+B-C))));
    return (2 * Math.PI * radius);
  }

  public static double rincircle (double A, double B, double C) {
    double s = (A + B + C) / 2;
    double radius = (Math.sqrt(((s - A) * (s - B ) * (s - C)) / s ));
    return radius;
  }

  public static double incircle (double A, double B, double C) {
    double s = (A + B + C) / 2;
    double radius = (Math.sqrt(((s - A) * (s - B ) * (s - C)) / s ));
    return (2 * Math.PI * radius);
  }


  //bisector lengths
  public static double bisectorlengthalfa (double A, double B, double C, double alfa, double beta, double charlie) {
    double p = 2 * B * C * Math.cos ( alfa / 2);
    double bis = p / (B + C);
    return bis;
  }
  public static double bisectorlengtbeta (double A, double B, double C, double alfa, double beta, double charlie) {
    double p = 2 * A * C * Math.cos ( beta / 2);
    double bis = p / (A + C);
    return bis;
  }
  public static double bisectorlengtcharlie (double A, double B, double C, double alfa, double beta, double charlie) {
    double p = 2 * A * B * Math.cos ( charlie / 2);
    double bis = p / (A + B);
    return bis;
  }
}
